function addFilesToQueue(files) {

document.getElementById('done-banner').classList.add('hidden');

if (!uploadRunning) runQueue();

async function runQueue() {
uploadRunning = true;
while (true) {
const item = uploadQueue.find(i=>i.status==='waiting');
if (!item) break;
item.status = 'processing';
renderQueue();
await processQueueItem(item);
renderQueue();
}
uploadRunning = false;
// Check if all done
const allDone = uploadQueue.every(i=>i.status==='done'||i.status==='manual');
if (allDone && uploadQueue.length>0) {
document.getElementById('done-banner').classList.remove('hidden');
}
}

async function processQueueItem(item) {
const key = localStorage.getItem('yoda_api_key');
if (!key) { item.status='manual'; item.note='אין API Key — נשמר לטיפול ידני'; saveQueueItem(item); return; }
try {
const base64 = await fileToBase64(item.file);
item._base64 = base64; // store for classify modal
const isPDF = item.file.type==='application/pdf';
const content = isPDF
? [{type:'document', source:{type:'base64', media_type:'application/pdf', data:base64}}]
: [{type:'image', source:{type:'base64', media_type:item.file.type, data:base64}}];

המספרים יהיו מספרים בלבד. תאריך בפורמט YYYY-MM-DD. אם שדה לא קיים השתמש ב-null.`});

const parsed = JSON.parse(text);
if (!parsed.supplier && !parsed.total) throw new Error('לא ניתן לזהות חשבונית');

// ===== PERIOD SELECTOR — builds buttons from actual invoice dates =====
// periods: 'month' = any single month, 'bimonth' = any 2-month pair, 'year' = any year
// selectedPeriod stored per report as { type:'month'|'bimonth'|'year', value:'2024-10'|'2024-10/2024-11'|'2024' }

// ===== PERIOD SELECTOR =====
let vatSelectedPeriod = null;
let plSelectedPeriod = null;
let ledgerSelectedPeriod = null;

function getApprovedInvoices() {
return getClientInvoices().filter(i => i.status === 'מאושר');
}
function getAvailableMonths() {
const months = new Set();
getApprovedInvoices().forEach(i => { if (i.date) months.add(i.date.slice(0,7)); });
return [...months].sort().reverse();
}
function getAvailableYears() {
const years = new Set();
getApprovedInvoices().forEach(i => { if (i.date) years.add(i.date.slice(0,4)); });
return [...years].sort().reverse();
}
function hebrewMonth(ym) {
const [y, m] = ym.split('-');
const names = ['','ינואר','פברואר','מרץ','אפריל','מאי','יוני','יולי','אוגוסט','ספטמבר','אוקטובר','נובמבר','דצמבר'];
return `${names[parseInt(m)]} ${y}`;
}

function buildPeriodBar(barId, sel, reportKey) {
const months = getAvailableMonths();
const years = getAvailableYears();
const bar = document.getElementById(barId);
if (!bar) return;
if (!months.length) { bar.innerHTML='<span style="color:var(--text3);font-size:13px">אין חשבוניות מאושרות</span>'; return; }

// Auto-select first month
if (!sel) {
if (reportKey==='vat') { vatSelectedPeriod = {type:'month',value:months[0]}; sel = vatSelectedPeriod; }
if (reportKey==='pl') { plSelectedPeriod = {type:'month',value:months[0]}; sel = plSelectedPeriod; }
if (reportKey==='ledger') { ledgerSelectedPeriod = {type:'month',value:months[0]}; sel = ledgerSelectedPeriod; }
}
sel = sel || {type:'month', value:months[0]};

const fn = reportKey==='vat' ? 'selectVatPeriod' : reportKey==='pl' ? 'selectPLPeriod' : 'selectLedgerPeriod';
const cls = (type, value) => `period-btn${sel.type===type && sel.value===value ? ' active' : ''}`;

const bimonths = [];
for (let i=0; i<months.length-1; i++) bimonths.push([months[i+1], months[i]]);

bar.innerHTML = `
<div style="display:flex;flex-direction:column;gap:8px;width:100%;margin-bottom:14px">
<div style="display:flex;gap:6px;flex-wrap:wrap;align-items:center">
<span style="color:var(--text3);font-size:12px;font-weight:600;min-width:52px">חודש:</span>
<div style="display:flex;gap:6px;flex-wrap:wrap">
${months.map(m=>`<button class="${cls('month',m)}" onclick="${fn}('month','${m}')">${hebrewMonth(m)}</button>`).join('')}
</div>
</div>
${bimonths.length ? `<div style="display:flex;gap:6px;flex-wrap:wrap;align-items:center">
<span style="color:var(--text3);font-size:12px;font-weight:600;min-width:52px">חודשיים:</span>
<div style="display:flex;gap:6px;flex-wrap:wrap">
${bimonths.map(([m1,m2])=>{const v=m1+'|'+m2;return`<button class="${cls('bimonth',v)}" onclick="${fn}('bimonth','${v}')">${hebrewMonth(m1)} – ${hebrewMonth(m2)}</button>`;}).join('')}
</div>
</div>` : ''}
${years.length ? `<div style="display:flex;gap:6px;flex-wrap:wrap;align-items:center">
<span style="color:var(--text3);font-size:12px;font-weight:600;min-width:52px">שנה:</span>
<div style="display:flex;gap:6px;flex-wrap:wrap">
${years.map(y=>`<button class="${cls('year',y)}" onclick="${fn}('year','${y}')">${y}</button>`).join('')}
</div>
</div>` : ''}
</div>`;
}

function filterBySelectedPeriod(invs, sel) {
if (!sel) return invs;
return invs.filter(i => {
const ym = i.date?.slice(0,7), y = i.date?.slice(0,4);
if (sel.type==='month') return ym === sel.value;
if (sel.type==='bimonth') { const [m1,m2]=sel.value.split('|'); return ym===m1||ym===m2; }
if (sel.type==='year') return y === sel.value;
return true;
});
}
